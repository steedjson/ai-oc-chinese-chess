### A. 测试代码示例

#### 异常检测器单元测试
```python
from games.services.anomaly_detector import AnomalyDetector
from django.utils import timezone
from datetime import timedelta

def test_timeout_detection():
    detector = AnomalyDetector()
    
    # 创建超时对局
    game = Game.objects.create(
        status='playing',
        started_at=timezone.now() - timedelta(hours=3)
    )
    
    anomalies = detector.detect_timeout_games()
    assert len(anomalies) > 0
    assert anomalies[0]['anomaly_type'] == 'timeout'
    assert anomalies[0]['severity'] == 'high'
```

#### API 测试
```python
from rest_framework.test import APITestCase
from django.contrib.auth import get_user_model

class GameManagementAPITest(APITestCase):
    def setUp(self):
        self.admin = get_user_model().objects.create_superuser(
            username='admin',
            password='admin123'
        )
        self.client.force_authenticate(user=self.admin)
    
    def test_abort_game(self):
        game = Game.objects.create(status='playing')
        response = self.client.post(
            f'/management/games/{game.id}/abort/',
            {'reason': '测试中止'}
        )
        self.assertEqual(response.status_code, 200)
        game.refresh_from_db()
        self.assertEqual(game.status, 'aborted')
```
